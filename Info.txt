Crear extension Unaccent en la base de datos Postgres
- CREATE EXTENSION IF NOT EXISTS unaccent WITH SCHEMA public;
- Para cada esquema, redirigir a la función pública
CREATE OR REPLACE FUNCTION rc.unaccent(text)
RETURNS text
LANGUAGE SQL
AS $$
  SELECT public.unaccent($1)
$$;
- Listar extensiones
SELECT extname, nspname AS schema_name
FROM pg_extension e
JOIN pg_namespace n ON e.extnamespace = n.oid;
- Mover extensión de esquema
ALTER EXTENSION unaccent SET SCHEMA public;
